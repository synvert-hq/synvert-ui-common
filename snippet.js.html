function filterSnippets(snippets, text) {
    const lowerCaseText = text.toLowerCase();
    return snippets.filter(snippet => (snippet.group.toLowerCase().includes(lowerCaseText) ||
        snippet.name.toLowerCase().includes(lowerCaseText) ||
        (!!snippet.description && snippet.description.toLowerCase().includes(lowerCaseText))));
}

function sortSnippets(snippets, text) {
    const lowerCaseText = text.toLowerCase();
    return snippets.sort((a, b) => {
        if (a.group.toLowerCase().includes(lowerCaseText) && !b.group.toLowerCase().includes(lowerCaseText))
            return -1;
        if (!a.group.toLowerCase().includes(lowerCaseText) && b.group.toLowerCase().includes(lowerCaseText))
            return 1;

        generatedSnippet += `  if_ruby '${rubyVersion}'\n`;
